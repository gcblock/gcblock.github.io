# 主要逻辑

调整以下两个参数对应相应数据参数，即可

  • ${parentId}  父ID
  • ${groupId}   ID
 arrayConvertTree (list = []) {
      const data = JSON.parse(JSON.stringify(list)) 
      const result = []
      if (!Array.isArray(data)) {
        return result
      }
      data.forEach(item => {
        delete item.children
      })
      const map = {}
      data.forEach(item => {
        map[item.groupId] = item
      })
      data.forEach(item => {
        const parent = map[item.parentId]
        if (parent) {
          (parent.children || (parent.children = [])).push(item)
        } else {
          result.push(item)
        }
      })
      return result
    }
